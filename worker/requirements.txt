# Pinned worker dependencies. Geometry, web service, data, validation.
# All external data sources are open or free tier. No paid data licences.

# Geospatial. The pipeline uses Shapely for geometry and delegates spatial
# candidate selection to PostGIS, so GeoPandas (and its heavy GDAL dependency)
# is not required at runtime. Add geopandas==1.0.1 here if a future stage needs
# dataframe-level geometry.
shapely==2.0.5
# Pin NumPy below 2.0: the Shapely 2.0.5 binary fails to build multi-geometries
# under NumPy 2.x ("ufunc 'create_collection' not supported"). Keep until Shapely
# is upgraded to a fully NumPy-2-compatible release.
numpy==1.26.4
# Coordinate transforms: OS grid reference (EPSG:27700) to WGS84.
pyproj==3.6.1

# Web service (worker exposes job submission and read endpoints to the API layer)
fastapi==0.111.1
uvicorn[standard]==0.30.3

# Database (Postgres with PostGIS)
psycopg[binary]==3.2.1
psycopg-pool==3.2.2
sqlalchemy==2.0.31
geoalchemy2==0.15.2

# Validation and settings
pydantic==2.8.2
pydantic-settings==2.3.4

# HTTP for geocode and isochrone providers (postcodes.io, OpenRouteService) and
# for server-side reference data downloads.
httpx==0.27.0
# Reading ONS income estimates spreadsheets during server-side reference loads.
openpyxl==3.1.5
# Reading legacy .xls workbooks (ONS HPSSA house prices). xlrd 2.x reads .xls.
xlrd==2.0.1

# Output generation. KML is emitted directly as XML (no native build
# dependency); PDF via reportlab; PPTX via python-pptx.
reportlab==4.2.2
python-pptx==1.0.2
